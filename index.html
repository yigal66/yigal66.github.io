<script>
const SVC    = '12345678-1234-1234-1234-123456789abc';
const C_CMD  = 'abcd1234-ab12-cd34-ef56-abcdef012345'; // write 0x01/0x02
const C_STAT = 'abcd1235-ab12-cd34-ef56-abcdef012345'; // notify LOCKED/UNLOCKED
const C_RSSI = 'abcd1236-ab12-cd34-ef56-abcdef012345'; // notify RSSI string

let cmdChar = null, device = null;

function log(msg, cls='inf') {
  const d = document.getElementById('log');
  const p = document.createElement('p');
  p.className = cls;
  p.textContent = '▸ ' + msg;
  d.appendChild(p);
  d.scrollTop = d.scrollHeight;
}

function setCarState(locked) {
  const card  = document.getElementById('stateCard');
  const emoji = document.getElementById('carEmoji');
  const label = document.getElementById('carLabel');
  card.className  = 'state-card ' + (locked ? 'locked' : 'unlocked');
  emoji.textContent = locked ? '🔒' : '🔓';
  label.className = 'car-label ' + (locked ? 'locked' : 'unlocked');
  label.textContent = locked ? 'LOCKED' : 'UNLOCKED';
}

function setRSSI(rssi) {
  document.getElementById('rssiVal').textContent = rssi + ' dBm';
  // Map -100..-30 dBm → 0..100%
  const pct = Math.max(0, Math.min(100, ((rssi - (-100)) / 70) * 100));
  document.getElementById('rssiFill').style.width = pct + '%';
}

async function connectBLE() {
  if (!navigator.bluetooth) {
    log('Web Bluetooth not available — enable chrome://flags → Experimental Web Platform Features', 'err');
    return;
  }
  try {
    log('Scanning for KugaLock...');
    device = await navigator.bluetooth.requestDevice({
      filters: [{ name: 'KugaLock' }],
      optionalServices: [SVC]
    });

    device.addEventListener('gattserverdisconnected', () => {
      log('Disconnected — reconnecting...', 'err');
      document.getElementById('btnConnect').className = '';
      document.getElementById('btnConnect').textContent = '🔵 Reconnect';
      cmdChar = null;
      setTimeout(reconnect, 2000);
    });

    await doConnect();

  } catch(e) {
    log('Error: ' + e.message, 'err');
  }
}

async function doConnect() {
  try {
    log('Connecting to ' + device.name + '...');
    const server  = await device.gatt.connect();
    const service = await server.getPrimaryService(SVC);

    cmdChar = await service.getCharacteristic(C_CMD);

    // Subscribe to lock state notifications
    const statChar = await service.getCharacteristic(C_STAT);
    await statChar.startNotifications();
    statChar.addEventListener('characteristicvaluechanged', e => {
      const val = new TextDecoder().decode(e.target.value);
      const locked = val.trim() === 'LOCKED';
      setCarState(locked);
      log('State: ' + val.trim(), locked ? 'inf' : 'ok');
    });

    // Subscribe to RSSI notifications
    const rssiChar = await service.getCharacteristic(C_RSSI);
    await rssiChar.startNotifications();
    rssiChar.addEventListener('characteristicvaluechanged', e => {
      const rssi = parseInt(new TextDecoder().decode(e.target.value));
      if (!isNaN(rssi)) setRSSI(rssi);
    });

    const btn = document.getElementById('btnConnect');
    btn.textContent = '✅ Connected to KugaLock';
    btn.className = 'connected';
    log('Connected! Proximity unlock active.', 'ok');

  } catch(e) {
    log('Connect failed: ' + e.message, 'err');
  }
}

async function reconnect() {
  if (device && !device.gatt.connected) {
    await doConnect().catch(() => setTimeout(reconnect, 3000));
  }
}

async function sendCmd(byte) {
  if (!cmdChar) { log('Not connected — tap Connect first', 'err'); return; }
  try {
    await cmdChar.writeValue(new Uint8Array([byte]));
    log(byte === 0x01 ? 'Manual UNLOCK sent' : 'Manual LOCK sent',
        byte === 0x01 ? 'ok' : 'inf');
  } catch(e) {
    log('Send error: ' + e.message, 'err');
  }
}
